57 /**
58  * @brief Manage data and its handler
59  *
60  * It is a data transport system with publish / subscribe semantics.
61  *
62  * # Internals #
63  * `std::list<T> mCurrentValues` is the holder for current values in the topic.
64  *
65  * This list is cleared once per evaluation at either:
66  * - The first Publish call during an evaluation pass (cleared
67  * before adding new value)
68  * - Topic is processed, and no further data will be Published to it
69  *
70  * The intended behavior is to have this list always contain all the
71  * data for a single evaluation pass - or nothing if that's the case.
72  */
73 template<class T>
74 class Topic : public BaseTopic
75 {
76  // Post-C++11 type checking
77 #if __cplusplus >= 201103L
78  static_assert(std::is_base_of<TopicState, T>::value, "T must inherit from TopicState");
79 #endif
80 
81 public:
82  Topic()
83  {
84  // Pre-C++11 type checking.
85 #if __cplusplus < 201103L
86  (void)static_cast<TopicState*>((T*)0);
87 #endif
88  }
89  /**
90  * @brief Append data to its list
91  */
92  void Publish(const T& arPayload)
93  {
94  if (Vertex::GetState() != kVertexProcessing)
95  {
96  mCurrentValues.clear();
97  Vertex::SetState(kVertexProcessing);
98  }
99 
100  mCurrentValues.push_back(arPayload);
101  }
102  /**
103  * @brief Pass pending data to its handler
104  */
105  void DispatchIntoSubscriber(SubscriberInterface<T>* aSubscriber)
106  {
107  if (mCurrentValues.size() > 0)
108  {
109  for (typename std::list<T>::iterator valueIt = mCurrentValues.begin();
110  valueIt != mCurrentValues.end();
111  ++valueIt)
112  {
113  aSubscriber->Evaluate(*valueIt);
114  }
115  }
116  }
117 
118  const std::list<T>& GetCurrentValues() const
119  {
120  return mCurrentValues;
121  }
122 
123  /**
124  * @brief Returns true if the new Data is available.
125  *
126  * New data is signaled by the Topic having been Processed and containing
127  * in storage.
128  */
129  inline bool HasNewValue() const
130  {
131  return (Vertex::GetState() == DetectorGraph::Vertex::kVertexDone &&
132  mCurrentValues.size() > 0);
133  }
134 
135  /**
136  * @brief Returns reference to the new/latest value in the topic.
137  *
138  * This method is a direct forwarding of the internal STL container and as
139  * such, calling GetNewValue on an empty Topic is undefined.
140  *
141  * On debug builds, a DG_ASSERT will check that the Topic contains a value.
142  */
143  inline const T& GetNewValue() const
144  {
145  DG_ASSERT(HasNewValue());
146  return mCurrentValues.back();
147  }
148 
149  virtual std::list<ptr::shared_ptr<const TopicState> > GetCurrentTopicStates() const
150  {
151  std::list<ptr::shared_ptr<const TopicState> > tCurrentTopicStates;
152 
153  for (typename std::list<T>::const_iterator valueIt = mCurrentValues.begin();
154  valueIt != mCurrentValues.end();
155  ++valueIt)
156  {
157  tCurrentTopicStates.push_back(ptr::shared_ptr<TopicState>(new T(*valueIt)));
158  }
159 
160  return tCurrentTopicStates;
161  }
162 
163  virtual TopicStateIdType GetId() const
164  {
165  return TopicState::GetId<T>();
166  }
167 
168  virtual ~Topic()
169  {}
170 
171  virtual void ProcessVertex()
172  {
173  // TODO(DGRAPH-5): This maintains the old detector graph state style
174  // if we want to preserve topic states until they're touched we can
175  // remove this clearing here.
176  if (Vertex::GetState() == kVertexClear)
177  {
178  mCurrentValues.clear();
179  }
180 
181  if (Vertex::GetState() == kVertexProcessing)
182  {
183  Vertex::SetState(kVertexDone);
184  MarkChildrenState(kVertexProcessing);
185  }
186  }
187 
188 private:
189  /*
190  * @brief List of current data in topic. @sa Topic.
191  */
192  std::list<T> mCurrentValues;
193 };
