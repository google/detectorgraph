93 bool TestTimeoutPublisherService::ForwardTimeAndEvaluate(TimeOffset aFwdTime, Graph& aGraphToEvaluate)
94 {
95  bool firedAtLeastOne = false;
96  TimeOffset finalDeadline = mElapsedTime + aFwdTime;
97 
98  // If we're forwarding any amount of time ahead
99  // flush input queue first.
100  if (aFwdTime > 0)
101  {
102  while (aGraphToEvaluate.HasDataPending())
103  {
104  aGraphToEvaluate.EvaluateGraph();
105  }
106  }
107 
108  MapIterator minIt = GetNextTimeout();
109 
110  while (minIt != mTimerMap.end())
111  {
112  TimeOffset nextDeadline = minIt->second;
113  if (nextDeadline > finalDeadline)
114  {
115  break;
116  }
117 
118  mElapsedTime = nextDeadline;
119 
120  if (minIt->first == mMetronomeId)
121  {
122  MetronomeFired();
123  SetTimeout(mMetronomeTimerPeriod, mMetronomeId);
124  Start(mMetronomeId);
125  }
126  else
127  {
128  TimeoutExpired(minIt->first);
129  mTimerMap.erase(minIt);
130  }
131 
132  firedAtLeastOne = true;
133 
134  while (aGraphToEvaluate.HasDataPending())
135  {
136  aGraphToEvaluate.EvaluateGraph();
137 
138  // If we're already at the targed time, evaluate only once and quit
139  // The idea is that finalDeadline is the "moment of interest" for a
140  // test; exiting here gives the test an opportunity inspect all
141  // outputs created for that particular moment in time.
142  if (nextDeadline == finalDeadline)
143  {
144  break;
145  }
146  }
147 
148  minIt = GetNextTimeout();
149  }
150 
151  mElapsedTime = finalDeadline;
152 
153  return firedAtLeastOne;
154 }
