64 /**
65  * @brief Produces a [Lagged<T>](@ref Lagged) for T
66  *
67  * This is a utility detector (and topic template) that performs a 1-lag on a
68  * TopicState. It's meant to enable closing feedback loops in a graph in an
69  * expressive and transparent way. It creates a Detector with the following
70  * relationships:
71  *
72  * @dot "Lag"
73 digraph GraphAnalyzer {
74  rankdir = "LR";
75  node[fontname=Helvetica];
76  size="12,5";
77  "T" [label="0:T",style=filled, shape=box, color=lightblue];
78  "T" -> "Lag";
79  "Lag" -> "Lagged" [style=dotted, color=red];
80  "Lag" [label="1:Lag<T>", color=blue];
81  "Lagged" [label="2:Lagged<T>",style=filled, shape=box, color=limegreen];
82 }
83  * @enddot
84  *
85  * Adding a Lag<T> Detector to a graph is done simply by adding it to a
86  * graph as shown ([from this example](@ref robotlocalization.cpp)) below:
87  @snippet robotlocalization.cpp RobotBrain Static
88  *
89  * Lagged<T> can then be used in any detector. In the same [example](
90  * @ref robotlocalization.cpp):
91  @snippet robotlocalization.cpp KalmanPoseCorrector Feedback Loop
92  *
93  */
94 template<class T>
95 class Lag : public Detector,
96  public SubscriberInterface< T >,
97  public FuturePublisher< Lagged< T > >
98 {
99 public:
100  Lag(Graph* graph) : Detector(graph)
101  {
102  Subscribe< T >(this);
103  SetupFuturePublishing< Lagged< T > >(this);
104  }
105 
106  virtual void Evaluate(const T& currentTopicState)
107  {
108  FuturePublisher< Lagged<T> >::PublishOnFutureEvaluation(Lagged<T>(currentTopicState));
109  }
110 };
