27 /**
28  * @brief A service that provides Timer function to DetectorGraph Detectors
29  *
30  * TimeoutPublisherService is used/shared among many TimeoutPublishers
31  * (Detectors) to add the notion of timed publications to DetectorGraph.
32  */
33 class TimeoutPublisherService
34 {
35 public:
36  /**
37  * @brief Constructor that initializes the service connected to a graph
38  *
39  * @param[in] graph The graph to which timed out TopicStates will be posted
40  */
41  TimeoutPublisherService(Graph& graph);

43  /**
44  * @brief Destructor
45  *
46  * Deletes all dynamically allocated pending TopicStates
47  */
48  virtual ~TimeoutPublisherService();

50  /**
51  * @brief Should return a unique id/handle for a new timer
52  *
53  * This must be implemented by subclasses. Different TimeoutPublishers will
54  * call this to 'acquire' a timer. The handle is then used throughout the API
55  * to refer to any individual timer
56  */
57  virtual TimeoutPublisherHandle GetUniqueTimerHandle() = 0;
58 
59  /**
60  * @brief Schedules a TopicState for Publishing after a timeout
61  *
62  * This is called internally by TimeoutPublishers. It starts a timer set to
63  * expire at a given deadline. When the deadline is reached @param aData is
64  * Published to the graph.
65  * Calling this method on an pending @param aTimerHandle resets it
66  * (canceling any previous timeouts)
67  *
68  * @param aData The TopicState to be published when the deadline expires.
69  * @param aMillisecondsFromNow The deadline relative to now.
70  * @param aTimerHandle A unique handle for this timer.
71  */
72  template<class T>
73  void ScheduleTimeout(const T& aData, const TimeOffset aMillisecondsFromNow, const TimeoutPublisherHandle aTimerHandle)
74  {
75  CancelPublishOnTimeout(aTimerHandle);
76  mScheduledTopicStatesMap[aTimerHandle] = new TopicStateDispatcher<T>(aData);
77  SetTimeout(aMillisecondsFromNow, aTimerHandle);
78  Start(aTimerHandle);
79  }
80 
81  /**
82  * @brief Should return the time offset to Epoch
83  *
84  * This must be implemented by subclasses. Different detectors may call
85  * this to acquire a timestamp - usually used to "stamp" a TopicState.
86  * This clock may jump back & forth due to time sync.
87  */
88  virtual TimeOffset GetTime() const = 0;
89 
90  /**
91  * @brief Should return monotonic time since some unspecified starting point.
92  *
93  * This must be implemented by subclasses.
94  * Returns the time offset to an unspecified point back in time that should
95  * not change for the duration of this instance.
96  * Different detectors may call this to acquire a consistent, strictly
97  * increasing, time offset valid for the duration of this object's instance.
98  */
99  virtual TimeOffset GetMonotonicTime() const = 0;
100 
101  /**
102  * @brief Cancels a timeout and deletes the stored TopicState
103  *
104  * This is called by different TimeoutPublishers when a timeout must be
105  * canceled.
106  */
107  void CancelPublishOnTimeout(const TimeoutPublisherHandle aTimerHandle);
108 
109  /**
110  * @brief Returns weather the timeout for a given handle has expired/fired
111  already.
112  *
113  * This will also return true if the referred timer never existed.
114  */
115  bool HasTimeoutExpired(const TimeoutPublisherHandle aTimerHandle) const;

117  /**
118  * @brief Schedules a TopicState for publishing periodically
119  *
120  * This is called by different Detectors with a TopicState and a publishing period.
121  * This method updates the metronome period based on the GCD of the requested publishing period.
122  * Calling 'StartPeriodicPublishing' will start publishing `T` to the graph periodically
123  * with interval @param aPeriodInMilliseconds .
124  *
125  * @param aPeriodInMilliseconds The regular period at which T should be published.
126  */
127  template<class T>
128  void SchedulePeriodicPublishing(const TimeOffset aPeriodInMilliseconds)
129  {
130  mMetronomePeriodMsec = gcd(aPeriodInMilliseconds, mMetronomePeriodMsec);
131  mScheduledPeriodicTopicStatesList.push_back(
132  PeriodicTopicStateDispatcher(aPeriodInMilliseconds, new TopicStateDispatcher<T>()));
133  }
134 
135  /**
136  * @brief Starts a Metronome to publish scheduled TopicStates
137  *
138  * Calling this method starts a metronome(periodic timer).
139  * TimeoutPublisherService will start publishing scheduled TopicStates periodically to graph.
140  */
141  void StartPeriodicPublishing();

183  void MetronomeFired();

200  /**
201  * @brief Internal DispatcherInterface for dispatching any scheduled
202  TopicState to Graph::PushData<T>
203  */
204  struct TopicStateDispatcherInterface
205  {
206  virtual void Dispatch(Graph& aGraph) = 0;
207  virtual ~TopicStateDispatcherInterface() {}

210  /**
211  * @brief Internal Dispatcher for dispatching a particular scheduled
212  TopicState to Graph::PushData<T>
213  */
214  template<class T>
215  struct TopicStateDispatcher : public TopicStateDispatcherInterface
216  {
217  TopicStateDispatcher() : mData() {}
218  TopicStateDispatcher(const T& aData) : mData(aData) {}
219  virtual void Dispatch(Graph& aGraph)
220  {
221  aGraph.PushData<T>(mData);

226  /**
227  * @brief Internal Dispatcher for periodically dispatching TopicState to Graph::PushData<T>

232  struct PeriodicTopicStateDispatcher

239  TopicStateDispatcherInterface* aTopicStateDispatcherInterface)

241  , mMetronomeCounter(0)
242  , mpTopicStateDispatcher(aTopicStateDispatcherInterface) {}
243  };
