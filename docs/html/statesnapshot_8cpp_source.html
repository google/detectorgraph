36 void StateSnapshot::UpdateValues(const std::list< ptr::shared_ptr<const TopicState> >& arTopicStates)
37 {
38  // Used for detection of consecutive/repeated named topics being published on the same evaluation
39  TopicStateIdType previousTopicStateID = TopicState::kAnonymousTopicState;
40 
41  typedef std::list< ptr::shared_ptr<const TopicState> >::const_iterator TopicStatePointerIterator;
42  for (TopicStatePointerIterator tDataIt = arTopicStates.begin();
43  tDataIt != arTopicStates.end();
44  ++tDataIt)
45  {
46  const ptr::shared_ptr<const TopicState> tpTopicState = *tDataIt;
47  if (tpTopicState->GetId() != TopicState::kAnonymousTopicState)
48  {
49  /* A malformed graph could publish more than one named TopicState
50  * with the same name on the same evaluation pass. This is not
51  * supported by StateSnapshot's map (and the second would just
52  * clobber the first one). In order to detect that bad design
53  * this check looks for consecutive TopicStates with matching
54  * names. The logic here relies on the fact that arTopicStates
55  * is composed by concatenating Topics one at a time so that
56  * repeated TopicStates would come consecutively.
57  * Thus if this check fails, multiple named TopicStates were
58  * published on same eval Pass */
59  if (tpTopicState->GetId() == previousTopicStateID)
60  {
61  // LCOV_EXCL_START
62  DG_LOG("!!!!!!!!!!!! Duplicate Detected (GetId = %d, previousTopicStateID = %d, Topic %s)",
63  (int)tpTopicState->GetId(), (int)previousTopicStateID, tpTopicState->GetName());
64  // LCOV_EXCL_STOP
65  } // LCOV_EXCL_LINE
66 
67  // Uncomment line below to re-enable paranoid check for development.
68  // DG_ASSERT(tpTopicState->GetId() != previousTopicStateID);
69 
70  previousTopicStateID = tpTopicState->GetId();
71 
72  // Just copying a ptr::shared_ptr around :)
73  mStateStore[tpTopicState->GetId()] = tpTopicState;
74  }
75  }
76 }
