25 /**
26  * @brief Push data to a topic when timer expires
27  *
28  * TimeoutPublisher provides a mechanism to schedule the publishing of a
29  * TopicState to the graph in the future. Similarly to FuturePublisher,
30  * the published TopicState goes into a separate (future) evaluation and so
31  * toposort constraints do not apply; this allows a detector to publish to
32  * itself.
33  *
34  * Additionally the detector can cancel/reset a scheduled job.
35  *
36  * Example:
37  * @code
38 class HeartbeatDetector : public Detector,
39  public SubscriberInterface<HeartBeat>,
40  public TimeoutPublisher<HeartBeat>
41 {
42  HeartbeatDetector(Graph* graph, TimeoutPublisherService* apTimeoutService) : Detector(graph)
43  {
44  Subscribe<HeartBeat>(this);
45  SetupTimeoutPublishing<HeartBeat>(this, apTimeoutService);
46  }
47 
48  void Evaluate(const HeartBeat&)
49  {
50  // This will be evaluated at 1Hz
51  PublishOnTimeout(HeartBeat(), 1000);
52  }
53 }
54  * @endcode
55  */
56 template<class T>
57 class TimeoutPublisher
58 {
59 public:
60  /**
61  * @brief Basic Constructor
62  *
63  * This constructor does not fully initialize the TimeoutPublisher as
64  * a TimeoutPublisherService is needed. That's done by SetTimeoutService.
65  */
66  TimeoutPublisher()
67  : mpTimeoutPublisherService(NULL), mDefaultHandle(0)
68  {
69  mDefaultHandle = 0;
70  }
71 
72  /**
73  * @brief Sets the timeout service and acquires a TimerHandle to be used
74  by the default/simple API.
75  */
76  void SetTimeoutService(TimeoutPublisherService* apTimeoutPublisherService)
77  {
78  mpTimeoutPublisherService = apTimeoutPublisherService;
79  mDefaultHandle = mpTimeoutPublisherService->GetUniqueTimerHandle();
80  }
81 
82  /**
83  * @brief Empty Virtual Destructor
84  */
85  virtual ~TimeoutPublisher() { }
86 
87  /**
88  * @brief Schedules a TopicState for Publishing after a timeout
89  *
90  * This method is analogous to Publish() in the sense that it's a Detector
91  * output but with two big differences:
92  * - @param aData is only published after @param aMillisecondsFromNow
93  * - @param aData is published to the Graph Input queue instead of another
94  * topic so TopoSort constraints do not apply; this allows a detector to
95  * publish and subscribe to the same topic.
96  * @param aTimerId (optional) allows detectors to control multiple concurrent timers.
97  * TimeoutPublisherHandles are vended through TimeoutPublisherService::GetUniqueTimerHandle
98  */
99  void PublishOnTimeout(const T& aData, const uint64_t aMillisecondsFromNow, TimeoutPublisherHandle aTimerId = 0)
100  {
101  if (aTimerId == 0) { aTimerId = mDefaultHandle; }
102 
103  mpTimeoutPublisherService->ScheduleTimeout<T>(aData, aMillisecondsFromNow, aTimerId);
104  }
105 
106  /**
107  * @brief Cancels the Scheduled PublishOnTimeout
108  */
109  void CancelPublishOnTimeout(TimeoutPublisherHandle aTimerId = 0)
110  {
111  if (aTimerId == 0) { aTimerId = mDefaultHandle; }
112 
113  mpTimeoutPublisherService->CancelPublishOnTimeout(aTimerId);
114  }
115 
116  /**
117  * @brief Returns weather a timeout has expired or not.
118  */
119  bool HasTimeoutExpired(TimeoutPublisherHandle aTimerId = 0) const
120  {
121  if (aTimerId == 0) { aTimerId = mDefaultHandle; }
122 
123  return mpTimeoutPublisherService->HasTimeoutExpired(aTimerId);
124  }
125 
126 protected:
127  // Keep reference to Graph in order to push data to graph in future
128  TimeoutPublisherService* mpTimeoutPublisherService;
129  TimeoutPublisherHandle mDefaultHandle;
130 };
